#include <iostream>
#include<vector>
#include<string>
#include<map>
#include<algorithm>
using namespace std;


int answer;
int main() {
	int answer = 0;
	int max_count = 0;
	for (int i = 2;i <= 1000;i++) {
		int cur = 10;
	
		int m_count = 0;
		map<int, int>m;
		while (1) {
			if (cur % i == 0) { //끝낫을때
				break;
			}
			else { //계손 반복됬을때
		
				int mok = cur / i;
				int rest = cur % i;
				m[mok]++;
				cur = rest*10;
				if (m[mok] == 3) {
					m[mok]--;
					for (auto it = m.begin();it != m.end();it++) {
						if (it->second == 2) {
							m_count++;
						}
					}
					break;//while문 탈출
				}
			}
		}
		cout << m_count << endl;
		if (max_count < m_count) {
			max_count= m_count;
			answer = i;
		}

	}
	cout << answer << endl;
}
